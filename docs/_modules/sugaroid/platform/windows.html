    def make_config():
        path = os.path.expanduser(os.path.join("~", "AppData", "Local", "sugaroid"))
        if not os.path.exists(path):
            try:
                os.makedirs(path)
            except Exception as e:
                logging.error(
                    "Error creating configuration file in dir {path}. Error code:{e}".format(
                        path=path, e=e
                    )
                )
        return path
